Marathon

I did this problem with complimentary counting. One record all possible skipped distances, and find the largest distance skipped and substract that from the total distance.

#include #include #include using namespace std; int N, counter; int dist(int x1, int x2, int y1, int y2) { return abs(x1 - x2) + abs(y1 - y2); } int distSkipped(int x1, int y1, int x2, int y2, int x3, int y3) { //printf("%d, %d, %d, %d, %d, %d\n", x1, y1, x2, y2, x3, y3); return dist(x1, x2, y1, y2) + dist(x2, x3, y2, y3) - dist(x1, x3, y1, y3); } int main() { ifstream fin("marathon.in"); fin >> N; int CheckPts[N][2]; for (counter = 0 ; counter < N ; counter++) { fin >> CheckPts[counter][0] >> CheckPts[counter][1]; } fin.close(); int totalDist = 0; for (counter = 0 ; counter < N - 1 ; counter++) { totalDist = totalDist + dist(CheckPts[counter][0], CheckPts[counter + 1][0], CheckPts[counter][1], CheckPts[counter + 1][1]); } //printf("%d\n", totalDist); int maxSkipped = distSkipped(CheckPts[0][0], CheckPts[0][1], CheckPts[1][0], CheckPts[1][1], CheckPts[2][0], CheckPts[2][1]); int distNow; //printf("%d\n", maxSkipped); for (counter = 2 ; counter < N - 2 ; counter++) { distNow = distSkipped(CheckPts[counter - 1][0], CheckPts[counter - 1][1], CheckPts[counter][0], CheckPts[counter][1], CheckPts[counter + 1][0], CheckPts[counter + 1][1]); if (distNow > maxSkipped) { maxSkipped = distNow; } //printf("%d\n", maxSkipped); } int minDist = totalDist - maxSkipped; ofstream fout("marathon.out"); fout << minDist << "\n"; fout.close(); return 0; }
